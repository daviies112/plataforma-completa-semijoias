## 3. Fine-Tuning de LLMs (OpenAI + Anthropic)

```python
# FINE-TUNING NO OPENAI (via API oficial)
import openai

# 1. PREPARAR DADOS (formato JSONL)
# MГѓВ­nimo: 10 exemplos. Recomendado: 50-100+ exemplos de qualidade
training_data = [
    {
        "messages": [
            {"role": "system", "content": "VocГѓВЄ ГѓВ© um assistente de suporte tГѓВ©cnico para a Antigravity Platform."},
            {"role": "user", "content": "Como reseto minha senha?"},
            {"role": "assistant", "content": "Para resetar sua senha na Antigravity Platform: 1. Clique em 'Esqueci minha senha' na tela de login. 2. Digite seu email cadastrado. 3. Verifique seu email (pode cair no spam). 4. Clique no link em atГѓВ© 1 hora. Se o problema persistir, contate suporte@antigravity.app"}
        ]
    },
    # ... mais exemplos
]

# Salvar como JSONL
import json
with open("training_data.jsonl", "w") as f:
    for example in training_data:
        f.write(json.dumps(example) + "\n")

# 2. UPLOAD E TREINAR
client = openai.OpenAI()

# Upload do arquivo
training_file = client.files.create(
    file=open("training_data.jsonl", "rb"),
    purpose="fine-tune"
)

# Criar job de fine-tuning
job = client.fine_tuning.jobs.create(
    training_file=training_file.id,
    model="gpt-4o-mini-2024-07-18",
    hyperparameters={
        "n_epochs": 3,
        "batch_size": 1,
        "learning_rate_multiplier": 1.8
    }
)

# 3. MONITORAR
# Verificar status:
status = client.fine_tuning.jobs.retrieve(job.id)
print(f"Status: {status.status}, Model: {status.fine_tuned_model}")

# 4. USAR O MODELO FINE-TUNED
response = client.chat.completions.create(
    model=status.fine_tuned_model,  # ex: ft:gpt-4o-mini:antigravity::AbcXyz
    messages=[{"role": "user", "content": "Como reseto minha senha?"}]
)
```

## 4. Deploy de Modelos Customizados

```python
# DEPLOY COM FASTAPI + DOCKER

from fastapi import FastAPI
from transformers import pipeline
import torch

app = FastAPI()

# Carregar modelo (uma vez no startup)
@app.on_event("startup")
async def load_model():
    global classifier
    classifier = pipeline(
        "text-classification",
        model="./modelo-fine-tuned",  # modelo local
        device=0 if torch.cuda.is_available() else -1
    )

@app.post("/predict")
async def predict(text: str):
    result = classifier(text)[0]
    return {
        "label": result["label"],
        "confidence": result["score"],
        "model_version": "v2.3"
    }

# Dockerfile para GPU
# FROM pytorch/pytorch:2.1.0-cuda11.8-cudnn8-runtime
```

```python
# DRIFT: quando o comportamento do modelo muda porque os dados mudaram
# Duas categorias: data drift (input mudou) e concept drift (relaГѓВ§ГѓВЈo input-output mudou)

from evidently import Report
from evidently.metric_preset import DataDriftPreset, TargetDriftPreset
import pandas as pd

# Comparar distribuiГѓВ§ГѓВЈo de dados de treino vs produГѓВ§ГѓВЈo
reference_data = pd.read_parquet("training_data.parquet")

report = Report(metrics=[DataDriftPreset(), TargetDriftPreset()])
report.run(reference_data=reference_data, current_data=current_data)

# Salvar relatГѓВіrio HTML
report.save_html("drift_report.html")

# Integrar com alertas:
drift_results = report.as_dict()
if drift_results["metrics"][0]["result"]["dataset_drift"]:
    notify_slack("ГўЕЎВ ГЇВёВЏ DRIFT DETECTADO Гўв‚¬вЂќ Considerar re-treino do modelo")
```
