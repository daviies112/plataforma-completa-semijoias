## 3. Rate Limiting com Redis

```python
from fastapi import Request, HTTPException
import redis
import time

redis_client = redis.Redis(host='localhost', port=6379, db=0)

async def rate_limit(request: Request, calls_per_minute: int = 60):
    """Rate limiting por IP + user com sliding window"""
    user_id = request.state.user.id if hasattr(request.state, 'user') else 'anonymous'
    key = f"rate_limit:{user_id}:{int(time.time() // 60)}"
    
    current = redis_client.incr(key)
    redis_client.expire(key, 60)  # expira em 1 minuto
    
    # Headers de rate limit (Stripe style Гўв‚¬вЂќ padrГѓВЈo de mercado)
    request.state.rate_limit_headers = {
        "X-RateLimit-Limit": str(calls_per_minute),
        "X-RateLimit-Remaining": str(max(0, calls_per_minute - current)),
        "X-RateLimit-Reset": str(int(time.time() // 60 + 1) * 60)
    }
    
    if current > calls_per_minute:
        raise HTTPException(
            status_code=429,
            detail="Rate limit exceeded",
            headers={
                **request.state.rate_limit_headers,
                "Retry-After": "60"
            }
        )
```

## 4. Webhooks: Design Correto

```python
# DESIGN DE WEBHOOK (como Stripe, GitHub fazem)

# 1. PAYLOAD PADRONIZADO
webhook_payload = {
    "id": "evt_abc123",           # ID ГѓВєnico do evento (para idempotГѓВЄncia)
    "type": "workflow.completed",  # namespace.aГѓВ§ГѓВЈo
    "created": 1708259400,         # Unix timestamp
    "livemode": True,              # ou false para sandbox
    "data": {
        "object": {
            "id": "wf_abc123",
            "name": "Instagram Monitor",
            "status": "completed",
            "execution_time_ms": 1234
        }
    }
}

# 2. ASSINATURA (HMAC-SHA256) Гўв‚¬вЂќ anti-spoofing
import hmac, hashlib, json

def sign_webhook(payload: dict, secret: str) -> str:
    body = json.dumps(payload, separators=(',', ':'))
    signature = hmac.new(
        secret.encode(),
        body.encode(),
        hashlib.sha256
    ).hexdigest()
    return f"sha256={signature}"

# Headers enviados:
# Webhook-Signature: sha256=abc123...
# Webhook-Timestamp: 1708259400
# Webhook-ID: evt_abc123

# 3. VERIFICAГѓвЂЎГѓЖ’O DO LADO DO CLIENTE
def verify_webhook(body: bytes, signature: str, secret: str, timestamp: str) -> bool:
    # Verificar timestamp (rejeitar eventos > 5 minutos velhos Гўв‚¬вЂќ replay attack)
    if abs(time.time() - int(timestamp)) > 300:
        return False
    
    expected = hmac.new(secret.encode(), body, hashlib.sha256).hexdigest()
    return hmac.compare_digest(f"sha256={expected}", signature)  # timing-safe
```
