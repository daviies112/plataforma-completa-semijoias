### Camada A ГўвӮ¬вҖқ Desktop/Web (Claude Computer Use API)
```python
# Para desktop Linux/Windows/Mac
import anthropic
import base64
from PIL import ImageGrab

client = anthropic.Anthropic()

def screenshot_to_base64():
    img = ImageGrab.grab()
    img.save("/tmp/screen.png")
    with open("/tmp/screen.png", "rb") as f:
        return base64.b64encode(f.read()).decode()

def computer_use_action(instruction: str):
    """
    Usa a Computer Use API da Anthropic para operar a tela
    """
    response = client.beta.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=4096,
        tools=[
            {
                "type": "computer_20250124",
                "name": "computer",
                "display_width_px": 1920,
                "display_height_px": 1080,
            },
            {"type": "text_editor_20250124", "name": "str_replace_editor"},
            {"type": "bash_20250124", "name": "bash"},
        ],
        messages=[
            {
                "role": "user",
                "content": [
                    {
                        "type": "image",
                        "source": {
                            "type": "base64",
                            "media_type": "image/png",
                            "data": screenshot_to_base64()
                        }
                    },
                    {"type": "text", "text": instruction}
                ]
            }
        ],
        betas=["computer-use-2025-01-24"]
    )
    return response
```

### Camada B ГўвӮ¬вҖқ Mobile (via Android Emulator/iOS Simulator)

```python
# Para ver e operar apps mobile (Instagram, WhatsApp, etc.)
# Usando: Android Debug Bridge (ADB) + screenshot

import subprocess

def mobile_screenshot():
    """Captura screenshot do Android emulador ou dispositivo conectado"""
    subprocess.run(["adb", "shell", "screencap", "-p", "/sdcard/screen.png"])
    subprocess.run(["adb", "pull", "/sdcard/screen.png", "/tmp/mobile_screen.png"])
    return "/tmp/mobile_screen.png"

def mobile_tap(x: int, y: int):
    """Toca em coordenadas no dispositivo Android"""
    subprocess.run(["adb", "shell", "input", "tap", str(x), str(y)])

def mobile_type(text: str):
    """Digita texto no dispositivo Android"""
    subprocess.run(["adb", "shell", "input", "text", text.replace(" ", "%s")])

def mobile_swipe(x1: int, y1: int, x2: int, y2: int):
    """Swipe no dispositivo Android"""
    subprocess.run(["adb", "shell", "input", "swipe", str(x1), str(y1), str(x2), str(y2)])
```
