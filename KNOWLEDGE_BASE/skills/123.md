import paramiko
import os

class VPSClient:
    def __init__(self):
        self.ssh = paramiko.SSHClient()
        self.ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
        self.ssh.connect(
            hostname=os.getenv("VPS_HOST"),
            username="root",
            key_filename=os.getenv("SSH_KEY_PATH")
        )
    
    def execute(self, command: str) -> dict:
        stdin, stdout, stderr = self.ssh.exec_command(command)
        return {
            "stdout": stdout.read().decode(),
            "stderr": stderr.read().decode(),
            "exit_code": stdout.channel.recv_exit_status()
        }
    
    def execute_safe(self, command: str) -> dict:
        """Executa somente comandos aprovados pelo Claude Guardian"""
        if is_dangerous_command(command):
            raise SafetyError(f"Comando nÃ£o permitido: {command}")
        return self.execute(command)
