# QUERY: estado atual = replay de todos os eventos
async def rebuild_order_state(order_id: str) -> Order:
    events = await db.fetch(
        "SELECT * FROM event_store WHERE aggregate_id = $1 ORDER BY timestamp",
        order_id
    )
    
    order = Order()
    for event in events:
        order.apply(deserialize_event(event))
    
    return order
