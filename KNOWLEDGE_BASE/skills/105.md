## 2. Fine-Tuning com LoRA/QLoRA (Eficiente em VRAM) ðŸ‹ï¸

```python
# Fine-tuning de Llama-3.1-8B com QLoRA
# Roda em GPU de 24GB (RTX 3090/4090 ou A10)

from transformers import AutoModelForCausalLM, AutoTokenizer, TrainingArguments
from peft import LoraConfig, get_peft_model, TaskType
from trl import SFTTrainer
import torch

# PASSO 1: Carregar modelo base com quantizaÃ§Ã£o 4-bit
model = AutoModelForCausalLM.from_pretrained(
    "meta-llama/Meta-Llama-3.1-8B-Instruct",
    load_in_4bit=True,         # QLoRA: 4-bit quantization
    bnb_4bit_compute_dtype=torch.bfloat16,
    device_map="auto"
)
tokenizer = AutoTokenizer.from_pretrained("meta-llama/Meta-Llama-3.1-8B-Instruct")

# PASSO 2: Configurar LoRA adapters
# Apenas ~1% dos parÃ¢metros sÃ£o treinados â€” eficiÃªncia mÃ¡xima
lora_config = LoraConfig(
    task_type=TaskType.CAUSAL_LM,
    r=16,                      # rank: 8-64 (maior = mais capacidade, mais VRAM)
    lora_alpha=32,             # scaling factor
    lora_dropout=0.1,
    target_modules=["q_proj", "v_proj", "k_proj", "o_proj"]
)

model = get_peft_model(model, lora_config)
model.print_trainable_parameters()
# Output: trainable params: 6,815,744 || all params: 8,037,224,448 (0.08% treinÃ¡veis)

# PASSO 3: Preparar dataset
from datasets import Dataset

# Formato de treino: instruction + input + output
training_data = [
    {
        "instruction": "Classifique o sentimento do texto como POSITIVO, NEGATIVO ou NEUTRO",
        "input": "O produto chegou antes do prazo e funcionou perfeitamente!",
        "output": "POSITIVO"
    },
    # ... mais exemplos
]

def format_prompt(example):
    return f"""### InstruÃ§Ã£o:
{example['instruction']}

### Entrada:
{example['input']}

### Resposta:
{example['output']}"""

dataset = Dataset.from_list(training_data).map(
    lambda x: {"text": format_prompt(x)}
)

# PASSO 4: Treinar
trainer = SFTTrainer(
    model=model,
    train_dataset=dataset,
    args=TrainingArguments(
        output_dir="./fine-tuned-model",
        num_train_epochs=3,
        per_device_train_batch_size=4,
        gradient_accumulation_steps=4,
        learning_rate=2e-4,
        fp16=True,
        save_strategy="epoch",
        logging_steps=10,
    ),
    dataset_text_field="text",
    max_seq_length=512,
)

trainer.train()
trainer.save_model("./fine-tuned-model")
```
