  private apiKey: string;
  private baseUrl: string;
  
  constructor(apiKey: string, options?: { baseUrl?: string }) {
    if (!apiKey) throw new SeuProdutoError('API key is required');
    this.apiKey = apiKey;
    this.baseUrl = options?.baseUrl ?? 'https://api.seuproduto.com/v1';
  }
  
  // MÃ©todo com tipagem completa e JSDoc:
  /**
   * Processa algo Ãºtil
   * @param input - Dados de entrada
   * @param options - OpÃ§Ãµes de processamento
   * @returns Promise com resultado
   * @throws SeuProdutoError se a requisiÃ§Ã£o falhar
   * @example

  async process(input: ProcessInput, options?: ProcessOptions): Promise<ProcessResult> {
    return this.request('POST', '/process', { input, ...options });
  }
  
  private async request<T>(method: string, path: string, body?: unknown): Promise<T> {
    const response = await fetch(`${this.baseUrl}${path}`, {
      method,
      headers: {
        'Authorization': `Bearer ${this.apiKey}`,
        'Content-Type': 'application/json',
        'User-Agent': `seuproduto-node/1.0.0`
      },
      body: body ? JSON.stringify(body) : undefined
    });
    
    if (!response.ok) {
      const error = await response.json();
      throw new SeuProdutoError(error.message, error.code, response.status);
    }
    
    return response.json();
  }
}
