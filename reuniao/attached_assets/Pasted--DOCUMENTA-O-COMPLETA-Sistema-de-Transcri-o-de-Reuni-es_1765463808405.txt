# 📋 DOCUMENTAÇÃO COMPLETA: Sistema de Transcrição de Reuniões 100ms + n8n

## 🎯 VISÃO GERAL DO SISTEMA

Este documento explica a integração completa entre sua plataforma Replit e o n8n para transcrição automática de reuniões usando 100ms.

### Fluxo Completo
```
1. Usuário agenda reunião → Sala 100ms criada → Dados salvos no Supabase
2. Usuário inicia reunião → Replit notifica n8n → n8n inicia transcrição HLS (sem gravar)
3. Usuário finaliza reunião → Replit notifica n8n → n8n busca transcrição → IA cria documento → Envia WhatsApp + Email
```

### Passo 2: Criar serviço de notificação (Replit)

Crie o arquivo: `server/services/n8nNotifications.js`

```javascript
// server/services/n8nNotifications.js

/**
 * Serviço para notificar o n8n sobre eventos de reunião
 */

const fetch = require('node-fetch');

/**
 * Notifica o n8n quando uma reunião é iniciada
 * @param {Object} dadosReuniao - Dados da reunião
 * @param {string} dadosReuniao.room_id - ID da sala 100ms
 * @param {string} dadosReuniao.nome - Nome do participante
 * @param {string} dadosReuniao.email - Email do participante
 * @param {string} dadosReuniao.telefone - Telefone do participante (formato: 5511999999999)
 */
async function notificarReuniaoIniciada(dadosReuniao) {
  try {
    const webhookUrl = process.env.N8N_WEBHOOK_REUNIAO_INICIADA;
    
    if (!webhookUrl) {
      throw new Error('N8N_WEBHOOK_REUNIAO_INICIADA não configurado');
    }

    const payload = {
      room_id: dadosReuniao.room_id,
      nome: dadosReuniao.nome,
      email: dadosReuniao.email,
      telefone: dadosReuniao.telefone,
      data_inicio: new Date().toISOString()
    };

    console.log('📤 Notificando n8n - Reunião iniciada:', payload);

    const response = await fetch(webhookUrl, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(payload)
    });

    if (!response.ok) {
      throw new Error(`Erro ao notificar n8n: ${response.status} ${response.statusText}`);
    }

    const result = await response.json();
    console.log('✅ n8n notificado com sucesso - Reunião iniciada');
    return result;

  } catch (error) {
    console.error('❌ Erro ao notificar reunião iniciada:', error);
    throw error;
  }
}

/**
 * Notifica o n8n quando uma reunião é finalizada
 * @param {string} roomId - ID da sala 100ms
 */
async function notificarReuniaoFinalizada(roomId) {
  try {
    const webhookUrl = process.env.N8N_WEBHOOK_REUNIAO_FINALIZADA;
    
    if (!webhookUrl) {
      throw new Error('N8N_WEBHOOK_REUNIAO_FINALIZADA não configurado');
    }

    const payload = {
      room_id: roomId,
      data_fim: new Date().toISOString()
    };

    console.log('📤 Notificando n8n - Reunião finalizada:', payload);

    const response = await fetch(webhookUrl, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(payload)
    });

    if (!response.ok) {
      throw new Error(`Erro ao notificar n8n: ${response.status} ${response.statusText}`);
    }

    const result = await response.json();
    console.log('✅ n8n notificado com sucesso - Reunião finalizada');
    return result;

  } catch (error) {
    console.error('❌ Erro ao notificar reunião finalizada:', error);
    throw error;
  }
}

module.exports = {
  notificarReuniaoIniciada,
  notificarReuniaoFinalizada
};
```

### Passo 3: Criar rotas para webhooks da 100ms (Replit)

Crie o arquivo: `server/routes/100msWebhooks.js`

```javascript
// server/routes/100msWebhooks.js

/**
 * Rotas para receber webhooks da 100ms
 * Documentação: https://www.100ms.live/docs/server-side/v2/features/webhooks
 */

const express = require('express');
const router = express.Router();
const { notificarReuniaoIniciada, notificarReuniaoFinalizada } = require('../services/n8nNotifications');
const { createClient } = require('@supabase/supabase-js');

// Inicializar Supabase
const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_KEY
);

/**
 * Webhook da 100ms para eventos de sala
 * A 100ms chama este endpoint quando:
 * - Alguém entra na sala (peer.join.success)
 * - Alguém sai da sala (peer.leave.success)
 * - A sala é criada ou encerrada
 */
router.post('/100ms-events', async (req, res) => {
  try {
    const event = req.body;
    
    console.log('📥 Webhook 100ms recebido:', event.type);

    // Evento: Primeiro participante entrou na sala
    if (event.type === 'peer.join.success') {
      const { room_id, peer } = event.data;

      // Verificar se é o primeiro participante (inicia a reunião)
      const { data: sala, error } = await supabase
        .from('reunioes')
        .select('*')
        .eq('room_id', room_id)
        .single();

      if (error) {
        console.error('Erro ao buscar sala:', error);
        return res.status(500).json({ error: error.message });
      }

      if (!sala) {
        return res.status(404).json({ error: 'Sala não encontrada' });
      }

      // Notificar n8n para iniciar transcrição
      await notificarReuniaoIniciada({
        room_id: room_id,
        nome: sala.nome,
        email: sala.email,
        telefone: sala.telefone
      });

      // Atualizar status da reunião para "em_andamento"
      await supabase
        .from('reunioes')
        .update({ status: 'em_andamento' })
        .eq('room_id', room_id);
    }

    // Evento: Último participante saiu da sala
    if (event.type === 'peer.leave.success') {
      const { room_id, peer } = event.data;

      // Verificar quantos participantes ainda estão na sala
      // Se for 0, finalizar a reunião
      const { data: peers, error } = await fetch(
        `https://api.100ms.live/v2/active-rooms/${room_id}/peers`,
        {
          headers: {
            'Authorization': `Bearer ${process.env.HMS_MANAGEMENT_TOKEN}`
          }
        }
      ).then(r => r.json());

      if (peers && peers.length === 0) {
        // Todos saíram, finalizar reunião
        await notificarReuniaoFinalizada(room_id);

        // Atualizar status da reunião para "concluida"
        await supabase
          .from('reunioes')
          .update({ 
            status: 'concluida',
            data_fim: new Date().toISOString()
          })
          .eq('room_id', room_id);
      }
    }

    res.json({ success: true });

  } catch (error) {
    console.error('Erro ao processar webhook 100ms:', error);
    res.status(500).json({ error: error.message });
  }
});

/**
 * Endpoint manual para iniciar transcrição
 * Usado se a 100ms não tiver webhooks configurados
 */
router.post('/iniciar-transcricao', async (req, res) => {
  try {
    const { room_id } = req.body;

    if (!room_id) {
      return res.status(400).json({ error: 'room_id é obrigatório' });
    }

    // Buscar dados da reunião no Supabase
    const { data: reuniao, error } = await supabase
      .from('reunioes')
      .select('*')
      .eq('room_id', room_id)
      .single();

    if (error || !reuniao) {
      return res.status(404).json({ error: 'Reunião não encontrada' });
    }

    // Notificar n8n
    await notificarReuniaoIniciada({
      room_id: room_id,
      nome: reuniao.nome,
      email: reuniao.email,
      telefone: reuniao.telefone
    });

    res.json({ success: true, message: 'Transcrição iniciada' });

  } catch (error) {
    console.error('Erro ao iniciar transcrição:', error);
    res.status(500).json({ error: error.message });
  }
});

/**
 * Endpoint manual para finalizar e transcrever reunião
 */
router.post('/finalizar-transcricao', async (req, res) => {
  try {
    const { room_id } = req.body;

    if (!room_id) {
      return res.status(400).json({ error: 'room_id é obrigatório' });
    }

    // Notificar n8n
    await notificarReuniaoFinalizada(room_id);

    // Atualizar status no Supabase
    await supabase
      .from('reunioes')
      .update({ 
        status: 'concluida',
        data_fim: new Date().toISOString()
      })
      .eq('room_id', room_id);

    res.json({ success: true, message: 'Reunião finalizada, transcrição em processamento' });

  } catch (error) {
    console.error('Erro ao finalizar transcrição:', error);
    res.status(500).json({ error: error.message });
  }
});

module.exports = router;
```

### Passo 5: Frontend - Botões para iniciar/finalizar transcrição

Adicione botões na interface da reunião: `src/pages/ReuniaoPage.jsx` (ou similar)

```jsx
// src/pages/ReuniaoPage.jsx

import { useState } from 'react';
import { Button } from '@/components/ui/button';

export default function ReuniaoPage() {
  const [roomId, setRoomId] = useState(null);
  const [transcricaoIniciada, setTranscricaoIniciada] = useState(false);

  // Quando a reunião iniciar (primeiro participante entrar)
  const handleReuniaoIniciada = async () => {
    try {
      const response = await fetch('/api/webhooks/iniciar-transcricao', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ room_id: roomId })
      });

      if (response.ok) {
        setTranscricaoIniciada(true);
        console.log('✅ Transcrição iniciada');
      }
    } catch (error) {
      console.error('Erro ao iniciar transcrição:', error);
    }
  };

  // Quando a reunião finalizar
  const handleReuniaoFinalizada = async () => {
    try {
      const response = await fetch('/api/webhooks/finalizar-transcricao', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ room_id: roomId })
      });

      if (response.ok) {
        console.log('✅ Reunião finalizada, documento sendo gerado...');
        alert('Reunião finalizada! Você receberá o documento estratégico por WhatsApp e Email em alguns minutos.');
      }
    } catch (error) {
      console.error('Erro ao finalizar reunião:', error);
    }
  };

  return (
    <div>
      {/* Sua interface de reunião aqui */}
      
      <div className="meeting-controls">
        {!transcricaoIniciada && (
          <Button onClick={handleReuniaoIniciada}>
            🎙️ Iniciar Transcrição
          </Button>
        )}
        
        {transcricaoIniciada && (
          <Button onClick={handleReuniaoFinalizada} variant="destructive">
            ⏹️ Finalizar Reunião e Gerar Documento
          </Button>
        )}
      </div>
    </div>
  );
}
```
