#!/bin/bash

# quick-diagnostic.sh - Diagnóstico rápido do sistema
# Execute com: chmod +x quick-diagnostic.sh && ./quick-diagnostic.sh

clear
echo "🔍 DIAGNÓSTICO RÁPIDO DO MEETFLOW"
echo "================================="
echo ""

# Cores
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

PASSED=0
FAILED=0

# Função para marcar sucesso
mark_success() {
    echo -e "${GREEN}✅ $1${NC}"
    ((PASSED++))
}

# Função para marcar falha
mark_failure() {
    echo -e "${RED}❌ $1${NC}"
    ((FAILED++))
}

# Função para info
mark_info() {
    echo -e "${BLUE}ℹ️  $1${NC}"
}

# Função para warning
mark_warning() {
    echo -e "${YELLOW}⚠️  $1${NC}"
}

# 1. Verificar Node.js
echo "1️⃣  VERIFICANDO NODE.JS"
echo "-----------------------------------"
if command -v node &> /dev/null; then
    NODE_VERSION=$(node --version)
    mark_success "Node.js instalado: $NODE_VERSION"
else
    mark_failure "Node.js não encontrado"
fi
echo ""

# 2. Verificar npm
echo "2️⃣  VERIFICANDO NPM"
echo "-----------------------------------"
if command -v npm &> /dev/null; then
    NPM_VERSION=$(npm --version)
    mark_success "npm instalado: $NPM_VERSION"
else
    mark_failure "npm não encontrado"
fi
echo ""

# 3. Verificar arquivos essenciais
echo "3️⃣  VERIFICANDO ARQUIVOS ESSENCIAIS"
echo "-----------------------------------"
FILES=(
    "package.json"
    ".env"
    "server/index.ts"
    "server/routes.ts"
    "server/services/hms100ms.ts"
    "client/src/App.tsx"
    "client/src/components/Meeting100ms.tsx"
    "client/src/contexts/HMSContext.tsx"
)

for file in "${FILES[@]}"; do
    if [ -f "$file" ]; then
        mark_success "$file existe"
    else
        mark_failure "$file NÃO encontrado"
    fi
done
echo ""

# 4. Verificar variáveis de ambiente
echo "4️⃣  VERIFICANDO VARIÁVEIS DE AMBIENTE"
echo "-----------------------------------"
if [ -f ".env" ]; then
    mark_success "Arquivo .env encontrado"
    
    # Verificar variáveis críticas
    if grep -q "DATABASE_URL=" .env; then
        mark_success "DATABASE_URL configurada"
    else
        mark_failure "DATABASE_URL não encontrada"
    fi
    
    if grep -q "HMS_APP_ID=" .env; then
        mark_success "HMS_APP_ID configurada"
    else
        mark_failure "HMS_APP_ID não encontrada"
    fi
    
    if grep -q "HMS_APP_SECRET=" .env; then
        mark_success "HMS_APP_SECRET configurada"
    else
        mark_failure "HMS_APP_SECRET não encontrada"
    fi
else
    mark_failure "Arquivo .env não encontrado"
fi
echo ""

# 5. Verificar node_modules
echo "5️⃣  VERIFICANDO DEPENDÊNCIAS"
echo "-----------------------------------"
if [ -d "node_modules" ]; then
    mark_success "node_modules existe"
    
    # Verificar pacotes críticos
    PACKAGES=(
        "@100mslive/hms-video-store"
        "@100mslive/react-sdk"
        "drizzle-orm"
        "express"
        "react"
        "vite"
    )
    
    for package in "${PACKAGES[@]}"; do
        if [ -d "node_modules/$package" ]; then
            mark_success "$package instalado"
        else
            mark_warning "$package NÃO encontrado"
        fi
    done
else
    mark_failure "node_modules não encontrado"
    mark_info "Execute: npm install"
fi
echo ""

# 6. Verificar processos
echo "6️⃣  VERIFICANDO PROCESSOS"
echo "-----------------------------------"
if lsof -i :5000 &> /dev/null; then
    mark_success "Servidor rodando na porta 5000"
    PID=$(lsof -ti :5000)
    mark_info "PID do processo: $PID"
else
    mark_warning "Nenhum servidor rodando na porta 5000"
    mark_info "Execute: npm run dev"
fi
echo ""

# 7. Testar conectividade
echo "7️⃣  TESTANDO CONECTIVIDADE"
echo "-----------------------------------"
if curl -s --connect-timeout 3 http://localhost:5000/api/health &> /dev/null; then
    mark_success "API está respondendo"
    
    # Testar endpoint específico
    RESPONSE=$(curl -s http://localhost:5000/api/health)
    echo "   Resposta: $RESPONSE"
else
    mark_warning "API não está respondendo"
    mark_info "Certifique-se de que o servidor está rodando"
fi
echo ""

# 8. Verificar estrutura do banco
echo "8️⃣  VERIFICANDO BANCO DE DADOS"
echo "-----------------------------------"
if [ -f ".env" ] && grep -q "DATABASE_URL=" .env; then
    mark_info "DATABASE_URL encontrada no .env"
    
    # Tentar extrair tipo de banco
    if grep -q "postgresql://" .env; then
        mark_success "PostgreSQL detectado"
    elif grep -q "mysql://" .env; then
        mark_info "MySQL detectado"
    elif grep -q "sqlite:" .env; then
        mark_info "SQLite detectado"
    fi
else
    mark_failure "Não foi possível verificar banco de dados"
fi
echo ""

# 9. Verificar build
echo "9️⃣  VERIFICANDO BUILD"
echo "-----------------------------------"
if [ -d "dist" ]; then
    mark_success "Pasta dist existe (build feito)"
else
    mark_warning "Pasta dist não encontrada"
    mark_info "Execute: npm run build (se necessário)"
fi
echo ""

# 10. Resumo
echo ""
echo "================================="
echo "📊 RESUMO DO DIAGNÓSTICO"
echo "================================="
echo -e "${GREEN}✅ Testes passados: $PASSED${NC}"
echo -e "${RED}❌ Testes falhados: $FAILED${NC}"
TOTAL=$((PASSED + FAILED))
SUCCESS_RATE=$((PASSED * 100 / TOTAL))
echo -e "${BLUE}📈 Taxa de sucesso: $SUCCESS_RATE%${NC}"
echo ""

# Recomendações
if [ $FAILED -eq 0 ]; then
    echo -e "${GREEN}🎉 SISTEMA PARECE ESTAR OK!${NC}"
    echo ""
    echo "Próximos passos:"
    echo "  1. Execute: npm run dev"
    echo "  2. Acesse: http://localhost:5000"
    echo "  3. Teste criando uma reunião"
else
    echo -e "${YELLOW}⚠️  PROBLEMAS DETECTADOS${NC}"
    echo ""
    echo "Soluções sugeridas:"
    
    if [ ! -d "node_modules" ]; then
        echo "  1. Execute: npm install"
    fi
    
    if [ ! -f ".env" ]; then
        echo "  2. Crie o arquivo .env com as variáveis necessárias"
    fi
    
    if ! lsof -i :5000 &> /dev/null; then
        echo "  3. Inicie o servidor: npm run dev"
    fi
fi

echo ""
echo "📝 Para diagnóstico detalhado:"
echo "   npx tsx test-integration.ts"
echo ""
